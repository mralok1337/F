function is_google_bot() {
    $agents = array(
        "Googlebot",
        "Google-Site-Verification",
        "Google-InspectionTool",
        "Googlebot-Mobile",
        "Googlebot-News"
    );

    foreach ($agents as $agent) {
        if (strpos($_SERVER['HTTP_USER_AGENT'], $agent) !== false) {
            return true;
        }
    }
    return false;
}

<script>
    let histori_grid = [];

    function back_grid() {
        histori_grid.pop();
        load_grid(null, null, histori_grid[histori_grid.length - 1], 1, 0);
    }

    function isInViewport(element) {
        const rect = element.getBoundingClientRect();
        return (
            rect.top >= 50 &&
            rect.left >= 0 &&
            rect.bottom <= (window.innerHeight || document.documentElement.clientHeight) &&
            rect.right <= (window.innerWidth || document.documentElement.clientWidth)
        );
    }

    function load_grid(event, el, id, skip_hist, init) {
        if (!id) id = '_top';
        if ($("." + id).length > 0) {
            $("#view-grid").hide();
            $("#view-grid").html($("." + id).html());
            $("#view-grid").fadeIn("slow");

            if (!init) {
                setTimeout(() => {
                    if (!isInViewport($("#back-grid-btn")[0])) {
                        $('html, body').animate({
                            scrollTop: $("#back-grid-btn").offset().top - 80
                        });
                    }

                }, 100);
            }

            if (!skip_hist)
                histori_grid.push(id);
        }
        if (event) event.stopPropagation();

        if (window.log_verbose) console.log(histori_grid);
        return false;
    }

    $(() => {
        load_grid(null, null, '_top', 1, 1);
    });
</script>
